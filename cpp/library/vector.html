Vector

#include <vector> // εισαγωγή βιβλιοθήκης

Ένα Vector είναι ενα διάνυσμα στοιχείων που μπορείτται να τα προσπελάσετε μέσω ενός δείκτη.
Δέχεται ΜΟΝΟ στοιχεία με τον τύπο στοιχεείου ποου έχει δηλωθεί.

vector<int> v={1,2,3,4,5,6} // vector με 6 στοιχεία int
vector<string> names = {"Diimitris","Maria"} // vector με 2 στοιχεία string
vector<int> vi(6) // vector με 6 στοιχεία int αρχικοποιημένα σε 0
vector<string> vname(6) // vector με 6 στοιχεία string αρχικοποιημένα σε ""

Εκτύπωση στοιχείων

Με το v.size() δίνει τον αριθμό των στοιχείων του vector μεε όνομα v

vector<int> v={1,2,3,4,5,6}
for (int i=0; i<v.size(); ++i){
    cout << v[i] << endl;
}

push_back()

Με το push_back(),μπορούμε να προσθέσουμε στο τέλος ένα νέο στοιχείο σε ένα vector.

\ 
#include <iostream>
#include <vector>

using namespace std; // Χρησιμοποιούμε το namespace std

int main() {
    // Δημιουργία vector ακεραίων
    vector<int> numbers;

    // Προσθήκη στοιχείων με τη χρήση του push_back
    numbers.push_back(10);
    numbers.push_back(20);
    numbers.push_back(30);

    // Εμφάνιση των στοιχείων του vector
    cout << "Τα στοιχεία του vector είναι: ";
    for (int i = 0; i < numbers.size(); ++i) {
        cout << numbers[i] << endl ;
    }
    return 0;
}
