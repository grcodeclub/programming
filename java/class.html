Το this χρησιμοποιείται για να αναφερθεί στις μεταβλητές του τρέχοντος αντικειμένου της κλάσης. Συγκεκριμένα όταν υπάρχει σύγκρουση ονομάτων μεταξύ των πεδίων της κλάσης και των τοπικών μεταβλητών ή παραμέτρων ενός μεθόδου, το this χρησιμοποιείται για να διαχωρίσει το πεδίο της κλάσης από τις τοπικές μεταβλητές. Έτσι, καθορίζεται ότι αναφερόμαστε στο πεδίο του αντικειμένου, όχι στην τοπική μεταβλητή.

public Vehicle(String make, String model, int year, double price) {
    // Η εκχώρηση της τιμής της παραμέτρου make στην μεταβλητή make του τρέχοντος αντικειμένου της κλάσης Vehicle.
    // Το this.make αναφέρεται στην μεταβλητή make του τρέχοντος αντικειμένου, ενώ το make αναφέρεται στην παράμετρο του κατασκευαστή.
    this.make = make;
    this.model = model;
    this.year = year;
    this.price = price;
}

Παράδειγμα

public class Vehicle {
    // Fields (Instance Variables)
    private String make;
    private String model;
    private int year;
    private double price;
    // Constructor (Κατασκευαστής)
    public Vehicle(String make, String model, int year, double price) {
        this.make = make;
        this.model = model;
        this.year = year;
        this.price = price;
    }
                    
    // Methods
    public String getMake() {
        return make;
    }
                    
    public String getModel() {
        return model;
    }
                    
    public int getYear() {
        return year;
    }
                    
    public double getPrice() {
        return price;
    }
                    
    // Main method (for testing the class)
    public static void main(String[] args) {
    // Creating instances of the class
    Vehicle vehicle1 = new Vehicle("Toyota", "Camry", 2022, 25000.0);
                    
    // Using methods of the class
    System.out.println("Vehicle 1:");
    System.out.println("Make: " + vehicle1.getMake());
    System.out.println("Model: " + vehicle1.getModel());
    System.out.println("Year: " + vehicle1.getYear());
    System.out.println("Price: $" + vehicle1.getPrice());
   
    }
}
